/** * Fungsi onOpen() berjalan otomatis saat Google Sheets dibuka. * Fungsi ini menambahkan menu khusus di toolbar (menu bar) atas. */ function onOpen() { var ui = SpreadsheetApp.getUi(); ui.createMenu('🔍 Alat Data') .addItem('Sorot Nama Ganda', 'sorotDuplikat') .addSeparator() // Garis pemisah .addItem('Hapus Semua Sorotan', 'hapusSorotan') .addToUi(); } /** * Fungsi utama untuk mencari dan memberikan warna pada data yang sama. * Script ini bekerja pada rentang sel (kolom) yang sedang Anda blok/pilih. */ function sorotDuplikat() { var sheet = SpreadsheetApp.getActiveSheet(); var range = sheet.getActiveRange(); var values = range.getValues(); var backgrounds = range.getBackgrounds(); var counts = {}; // Anda bisa mengganti kode warna hex di bawah ini (contoh: "#FFCCCC" adalah merah muda) var warnaSorot = "#FFCCCC"; var warnaNormal = null; // null berarti menghapus warna/transparan // Langkah 1: Menghitung berapa kali setiap nama muncul di area yang dipilih for (var i = 0; i < values.length; i++) { for (var j = 0; j < values[i].length; j++) { // Mengubah nilai menjadi huruf kecil dan menghapus spasi ekstra agar pencocokan akurat var cellValue = values[i][j].toString().trim().toLowerCase(); if (cellValue !== "") { if (counts[cellValue]) { counts[cellValue]++; } else { counts[cellValue] = 1; } } } } // Langkah 2: Menandai latar belakang sel jika namanya muncul lebih dari 1 kali for (var i = 0; i < values.length; i++) { for (var j = 0; j < values[i].length; j++) { var cellValue = values[i][j].toString().trim().toLowerCase(); if (cellValue !== "" && counts[cellValue] > 1) { backgrounds[i][j] = warnaSorot; } else { backgrounds[i][j] = warnaNormal; // Mengembalikan ke tanpa warna jika unik } } } // Menerapkan perubahan warna ke dalam sheet sekaligus (agar prosesnya cepat) range.setBackgrounds(backgrounds); } /** * Fungsi tambahan untuk membersihkan/menghapus warna sorotan di sel yang dipilih */ function hapusSorotan() { var sheet = SpreadsheetApp.getActiveSheet(); var range = sheet.getActiveRange(); // Mengatur latar belakang menjadi null (tanpa warna) range.setBackground(null); }
